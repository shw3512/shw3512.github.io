\begin{enumerate}[label=(\alph*)]
\item\label{itm : LQ09B a} (2 pt) Show that $\displaystyle\lim_{x \rightarrow 0} \frac{x^{2} + 2 x + 2 - 2 e^{x}}{x^{3}} = -\frac{1}{3}$.

\fontHint{While not required, you may use the Taylor series $e^{x} = 1 + x + \frac{1}{2} x^{2} + \frac{1}{6} x^{3} + \ldots$.}
\end{enumerate}
