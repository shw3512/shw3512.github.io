\section{Exercise}

(5 pt) Evaluate the following integral.
\begin{align*}
\int \sin^{3} t \spaceIntd \intd t.
\end{align*}
\fontHint{View the integral as $\int (\sin^{2} t) (\sin t) \intd t$. After an integration, use the identity $\sin^{2} t + \cos^{2} t = 1$. You should get an equation involving the given integral on both the left and the right. Now treat the integral like an unknown variable (e.g., $x$) and solve for it.}

\spaceSolution{6in}{% Begin solution.
Using integration by parts with
\begin{align*}
u
=
\sin^{2} t,
&&
\intd v
=
\sin t \spaceIntd \intd t,
\end{align*}
so that
\begin{align*}
\intd u
=
2 \sin t \cos t \spaceIntd \intd t,
&&
v
=
-\cos t,
\end{align*}
we get
\begin{align*}
\int \sin^{3} t \spaceIntd \intd t
=
-\sin^{2} t \cos t + 2 \int \sin t \cos^{2} t \spaceIntd \intd t.
\end{align*}
Using the identity $\sin^{2} t + \cos^{2} t = 1$, we can write this last expression as
\begin{align*}
\int \sin^{3} t \spaceIntd \intd t
&=
-\sin^{2} t \cos t + 2 \int \sin t \left(1 - \sin^{2} t\right) \spaceIntd \intd t
\\
&=
-\sin^{2} t \cos t + 2 \int \sin t \spaceIntd \intd t - 2 \int \sin^{3} t \spaceIntd \intd t.
\end{align*}
Solving for $\int \sin^{3} t \spaceIntd \intd t$, we find
\begin{align*}
\int \sin^{3} t \spaceIntd \intd t
=
-\frac{1}{3} \sin^{2} t \cos t - \frac{2}{3} \cos t + c,
\end{align*}
where $c \in \reals$.}% End solution.
