\section{Exercise}

(5 pt) Evaluate the following integral.
\begin{align*}
\int t^{3} \ln(t^{2}) \spaceIntd \intd t.
\end{align*}

\spaceSolution{6in}{% Begin solution.
Using integration by parts, we compute
\begin{align*}
\int t^{3} \ln(t^{2}) \spaceIntd \intd t
=
\frac{1}{4} t^{4} \ln(t^{2}) - \frac{1}{8} t^{4} + c,
\end{align*}
where $c \in \reals$.}% End solution.
