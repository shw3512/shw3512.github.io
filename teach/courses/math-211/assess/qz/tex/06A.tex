\section{Exercise}

(5 pt) Find the general solution to the nonhomogeneous linear ODE
\begin{align*}
y' - 3 y
=
-10 \sin t.
\end{align*}

\spaceSolution{6in}{% Begin solution.
The general solution is
\begin{align*}
y(t)
=
3 \sin t + \cos t + c e^{3 t},
\end{align*}
where $c \in \reals$.}% End solution.
