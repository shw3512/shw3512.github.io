\section{Exercise}

(5 pt) Find the general solution to the nonhomogeneous 1st-order linear ODE
\begin{align*}
2 y' + 4 y
=
8 \sin(2 t).
\end{align*}

\spaceSolution{6in}{% Begin solution.
The general solution is
\begin{align*}
y(t)
=
\sin(2 t) - \cos(2 t) + c e^{-2 t},
\end{align*}
where $c \in \reals$.}% End solution.
