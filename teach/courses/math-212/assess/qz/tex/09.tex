\section{Exercise}

(2 pt) Consider the function $f: \reals^{2} \rightarrow \reals$ given by
\begin{align*}
f(x,y)
=
x^{2} + \frac{y^{2}}{4} + 1.
\end{align*}
\begin{enumerate}[label=(\alph*)]
\item (1 pt) For what values of $c$ is the level set $LS_{f}(c)$ empty? \fontHint{What output values can $f$ \emph{not} give?}
\end{enumerate}

\spaceSolution{.1in}{}

\begin{enumerate}[resume,label=(\alph*)]
\item (1 pt) Draw three nonempty level sets $LS_{f}(c)$ of $f$ in the $x y$-plane, labeled with their corresponding values $c$.
\end{enumerate}
