      \let\oldc\c%

   }{}%
   \foreach \s in {0,...,\maxNumSections}{%
      \ifthenelse{\s<10}{%
         \let\olds\s%
         \renewcommand{\s}{0\olds}%
      }{}%
%      \filePathChapters Chapter\c/\c\s%
      \IfFileExists{\filePathChapters Chapter\c/\c_\s}{%
         \input{\filePathChapters Chapter\c/\c_\s}%
      }{}% If file does not exist, do nothing.
   }%
}%
