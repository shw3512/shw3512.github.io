% Dummit & Foote 3e Exercises 7.3.29 and 30
\noindent{}Let $R$ be a commutative ring with a multiplicative identity $1_{R} \neq 0_{R}$. An element $a \in R$ is \fontDefWord{nilpotent} if there exists an $n \in \integersPositive$ such that $a^{n} = 0_{R}$. The \fontDefWord{nilradical} of $R$ is the set
\begin{align*}
\nilradical(R)
=
\{a \in R \st \text{$a$ is nilpotent}\}
\end{align*}
\begin{enumerate}[label=(\alph*)]
\item\label{itm : LQ01Ca} Prove that $\nilradical(R)$ is an ideal of $R$.
\item\label{itm : LQ01Cb} Prove that $\nilradical(R / \nilradical(R)) = \{0\}$.
\end{enumerate}
